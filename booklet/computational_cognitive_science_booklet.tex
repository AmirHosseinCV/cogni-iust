\chapter{چارچوب درس و مسئلهٔ اصلی}

\section{هدف درس}
موضوع درس، \term{علوم شناختی محاسباتی}{Computational Cognitive Science} است. علوم شناختی می‌پرسد انسان چگونه می‌بیند، می‌شنود، حرف می‌زند، فکر می‌کند، تصمیم می‌گیرد، خطا می‌کند و یاد می‌گیرد. علوم شناختی محاسباتی یک گام جلوتر می‌رود و می‌پرسد چگونه می‌توان این فرایندها را به‌صورت محاسباتی، صوری و قابل پیاده‌سازی مدل کرد.

در این درس، تمرکز اصلی روی پیوند میان علوم شناختی و هوش مصنوعی است؛ به‌ویژه:
\begin{itemize}
  \item \term{پردازش زبان طبیعی}{NLP}
  \item \term{استدلال}{Reasoning}
  \item \term{پردازش گفتار}{Speech Processing}
  \item \term{مدل‌های زبانی بزرگ}{Large Language Models / LLMs}
  \item \term{ترنسفورمر}{Transformer}
\end{itemize}

مسیر کلی درس این است که ابتدا یک نظریهٔ شناختی را بفهمیم، سپس ببینیم مدل‌های یادگیری ماشین یا یادگیری عمیق در همان مهارت چه محدودیتی دارند، و در نهایت بپرسیم آیا آن نظریهٔ شناختی می‌تواند ایده‌ای برای بهبود مدل بدهد یا نه.

\section{تفاوت پرسش شناختی و پرسش یادگیری ماشین}
در یادگیری ماشین، پرسش معمولاً چنین است:
\[
\text{چگونه این \lr{task} را با \lr{accuracy} بهتر حل کنیم؟}
\]
اما در علوم شناختی، پرسش اصلی این است:
\[
\text{انسان چگونه این \lr{task} را انجام می‌دهد؟}
\]

اگر یک مدل خروجی درست بدهد، هنوز لزوماً مدل شناختی خوبی نیست. مدل شناختی باید بتواند فرایند، خطاها، تفاوت‌های فردی و گاهی اختلالات انسانی را هم توضیح دهد. به همین دلیل در علوم شناختی، فقط نتیجه مهم نیست؛ مسیر رسیدن به نتیجه نیز اهمیت دارد.

\section{نظریه‌های شناختی به‌عنوان لمس بخشی از یک فیل}
استاد برای توضیح تعدد نظریه‌های شناختی از استعارهٔ فیل استفاده کرد: هر نظریه بخشی از پدیدهٔ شناخت انسان را توضیح می‌دهد. یک نظریه ممکن است شنیدن را توضیح دهد، دیگری حافظه را، دیگری توجه را و دیگری سوگیری شناختی را. هیچ نظریه‌ای به‌تنهایی کل ذهن انسان را توضیح نمی‌دهد.

یک نظریهٔ شناختی خوب باید بتواند:
\begin{itemize}
  \item رفتار درست و خطاهای انسان را توضیح دهد.
  \item فقط مخصوص یک گروه محدود نباشد.
  \item دربارهٔ کودک، سالمند، فرد سالم، فرد دارای اختلال و افراد با توانایی‌های متفاوت حرفی برای گفتن داشته باشد.
  \item آزمایش‌پذیر باشد و پیش‌بینی تولید کند.
\end{itemize}

\section{از هوش مصنوعی تا علوم شناختی محاسباتی}
مسیر دانشی درس را می‌توان چنین دید:
\[
\text{\lr{AI}} \rightarrow \text{\lr{Machine Learning}} \rightarrow \text{\lr{Deep Learning}} \rightarrow \text{\lr{Computational Cognitive Science}}
\]

در \eng{AI} کلاسیک با جست‌وجو، برنامه‌ریزی، الگوریتم‌های ژنتیک و روش‌هایی مثل \eng{BFS} یا \eng{simulated annealing} روبه‌رو می‌شویم. در یادگیری ماشین، مدل از داده یاد می‌گیرد. در یادگیری عمیق، بازنمایی‌های پیچیده‌تر و معماری‌های عمیق‌تر ساخته می‌شوند. در علوم شناختی محاسباتی می‌پرسیم: کدام مهارت شناختی در این مدل‌ها غایب است و چگونه می‌توان آن را اضافه کرد؟

\section{مطالعات کلاسیک، مطالعات مدرن و کیس‌استادی درس}
در کلاس توضیح داده شد که بسیاری از درس‌های رایج علوم شناختی محاسباتی، به‌ویژه در کورس‌های دانشگاهی، مدت زیادی روی \eng{vision} و پردازش تصویر تمرکز داشته‌اند؛ چون هم داده و آزمایش‌های کلاسیک زیادی در این حوزه وجود دارد و هم تاریخ هوش مصنوعی نیز رشد مهمی را از بینایی ماشین تجربه کرده است. در این درس، کیس‌استادی اصلی بیشتر به زبان، گفتار، استدلال و مدل‌های زبانی نزدیک شد تا ارتباط مستقیم‌تری با دانشجویان هوش مصنوعی داشته باشد.

منابع درس دو جنس دارند:
\begin{itemize}
  \item مقاله‌ها و نظریه‌های کلاسیک شناختی، مثل مدل‌های تشخیص گفتار، حافظه، توجه و سوگیری.
  \item مقاله‌های مدرن هوش مصنوعی که یک ضعف شناختی در مدل‌های عمیق یا مدل‌های زبانی را نشان می‌دهند و سپس برای آن راه‌حل محاسباتی پیشنهاد می‌کنند.
\end{itemize}

بنابراین خواندن مقاله در این درس فقط برای دانستن یک روش فنی نیست؛ هدف این است که دانشجو بتواند بین نظریهٔ شناختی، طراحی آزمایش، دادهٔ انسانی و معماری محاسباتی پل بزند.

\section{ارزیابی شناختی مدل‌های بزرگ}
یکی از بحث‌های مهم جلسهٔ اول این بود که مدل‌های بزرگ ممکن است در \eng{benchmark}های معمول خوب عمل کنند، اما در مهارت‌هایی مثل قیاس، تعمیم انتزاعی و \eng{counterfactual reasoning} شکست بخورند. مثال کلاس، آزمون‌های قیاسی با الفبای ساختگی بود: اگر مدل فقط ترتیب الفبای آشنا را حفظ کرده باشد، با تغییر نمادها یا برعکس کردن ترتیب، ممکن است نتواند مفهوم انتزاعی «بعدی بودن» را منتقل کند.

در آزمون‌های قیاس داستانی نیز ممکن است دو داستان از نظر سطحی بسیار متفاوت باشند، اما \eng{theme} مشترکی داشته باشند؛ مثلاً هر دو دربارهٔ کمک کردن باشند. انسان معمولاً می‌تواند قیاس‌های دورتر را از راه رابطهٔ عمیق‌تر بفهمد، اما مدل ممکن است فقط شباهت‌های سطحی را دنبال کند.

برای ارزیابی شناختی مدل، دو معیار جدا باید بررسی شود:
\begin{itemize}
  \item آیا مدل \eng{task} را درست انجام می‌دهد؟
  \item آیا الگوی خطا و سختی مدل با الگوی خطا و سختی انسان \eng{correlation} دارد؟
\end{itemize}

در همین چارچوب، استاد به بحث \term{هوش مصنوعی عمومی}{Artificial General Intelligence} و دسته‌بندی مهارت‌های شناختی نیز اشاره کرد. چارچوب‌هایی مثل \eng{Cattell-Horn-Carroll} می‌توانند برای تعریف انواع توانایی شناختی و طراحی آزمون‌های مقایسه‌ای بین انسان و مدل استفاده شوند.

اگر هدف ساختن یک ابزار بهتر از انسان باشد، می‌خواهیم خطاهای انسانی را کاهش دهیم. اما اگر هدف مدل‌سازی شناخت انسان باشد، شباهت الگوی خطا اهمیت مرکزی دارد. مسیر پژوهشی پیشنهادی کلاس این بود: نخست نشان دهیم انسان و مدل در یک خطای شناختی مشترک‌اند، سپس با نظریهٔ شناختی آن خطا را توضیح دهیم و در نهایت با \eng{fine-tuning}، \eng{prompting}، \eng{RAG} یا تغییر معماری آن را هدفمند کاهش دهیم.

\begin{assignmentbox}{نکات اجرایی جلسهٔ اول}
\begin{itemize}
  \item دانشجویان باید در گروه \eng{Bale} درس عضو شوند.
  \item هر دانشجو باید یک بیوگرافی یا رزومهٔ کوتاه برای استاد ارسال کند: دانشگاه قبلی، معدل، استاد راهنما، خط پژوهشی، سابقهٔ پژوهشی و علایق مرتبط.
  \item فایل‌های \eng{Excalidraw} هر جلسه در اختیار دانشجویان قرار می‌گیرد و مبنای پیگیری تصویری کلاس است.
\end{itemize}
\end{assignmentbox}

\chapter{نظریه‌های اولیهٔ زبان و اعتبارسنجی شناختی}

\section{\eng{Cohort Model}: از صدا به کلمه}
\term{مدل Cohort}{Cohort Model} یکی از نظریه‌های کلاسیک در تشخیص کلمهٔ گفتاری است. ایدهٔ اصلی آن این است که وقتی بخشی از یک کلمه را می‌شنویم، مجموعه‌ای از کلمات ممکن در ذهن فعال می‌شوند. هرچه صدای بیشتری دریافت می‌شود، گزینه‌های ناسازگار حذف می‌شوند تا در نهایت کلمهٔ مناسب انتخاب شود.

مثلاً در رستوران، اگر صدایی با «بـ» آغاز شود، گزینه‌هایی مثل «برنج» یا «بریانی» فعال می‌شوند. اگر صدای بعدی «ر» باشد، لیست محدودتر می‌شود. این فرایند شبیه پیشنهادهای تکمیل خودکار است، اما در ذهن انسان رخ می‌دهد.

برای فهم بهتر، \eng{cohort} را می‌توان «گروه نامزدها» دانست. در لحظهٔ شنیدن اولین بخش صوتی، ذهن هنوز نمی‌داند کلمهٔ نهایی چیست، پس چند نامزد را فعال نگه می‌دارد. با رسیدن هر قطعهٔ صوتی جدید، نامزدهایی که دیگر با ورودی سازگار نیستند حذف می‌شوند. بنابراین مدل توضیح می‌دهد چرا ابتدای کلمه در تشخیص گفتار بسیار مهم است و چرا نویز یا ابهام در همان ابتدا می‌تواند مسیر انتخاب کلمه را عوض کند.

\begin{definitionbox}{ویژگی اصلی \eng{Cohort}}
جریان اطلاعات عمدتاً \term{پایین‌به‌بالا}{bottom-up} است:
\[
\text{صدا} \rightarrow \text{کلمات ممکن} \rightarrow \text{انتخاب کلمه}
\]
\end{definitionbox}

محدودیت این مدل آن است که انسان همیشه مثل یک \eng{autocomplete} ساده عمل نمی‌کند. اگر فقط جریان پایین‌به‌بالا وجود داشته باشد، انتظار قبلی، موقعیت، موضوع گفت‌وگو و دانش زبانی نباید نقش زیادی داشته باشند؛ در حالی که در تجربهٔ واقعی، همین عوامل می‌توانند شنیدن یک صدای مبهم را تغییر دهند.

\section{\eng{TRACE}: نقش انتظار و \eng{context}}
مدل \term{TRACE}{TRACE Model} دیدگاه تعاملی‌تری ارائه می‌کند. در این مدل، فقط صدا تعیین‌کننده نیست؛ انتظار، \eng{context} و دانش قبلی نیز روی تفسیر صدا اثر می‌گذارند. اگر بخشی از صدا مبهم باشد، ذهن می‌تواند با توجه به \eng{context} آن را کامل کند.

تفاوت اصلی:
\begin{itemize}
  \item در \eng{Cohort}: آنچه می‌شنویم، گزینه‌ها را محدود می‌کند.
  \item در \eng{TRACE}: آنچه می‌شنویم و آنچه انتظار داریم بشنویم، هم‌زمان اثر می‌گذارند.
\end{itemize}

پس \eng{TRACE} صرفاً رد کردن \eng{Cohort} نیست؛ بیشتر یک گام تکاملی است. \eng{Cohort} جنبهٔ مهمی از تشخیص گفتار را می‌گیرد، اما \eng{TRACE} نشان می‌دهد پردازش گفتار تعاملی است: اطلاعات صوتی از پایین به بالا می‌آید و انتظار معنایی/زبانی از بالا به پایین روی تفسیر آن اثر می‌گذارد. این همان الگوی کلی بسیاری از نظریه‌های شناختی است: نظریهٔ قبلی بخشی از پدیده را خوب توضیح می‌دهد، سپس نظریهٔ جدید تلاش می‌کند بخش‌های جاافتاده را اضافه کند.

\section{اعتبارسنجی نظریه‌های شناختی}
یک نظریهٔ شناختی فقط با مثال قانع‌کننده معتبر نمی‌شود. باید با آزمایش‌های مستقل، دادهٔ انسانی، تحلیل آماری و گاهی دادهٔ مغزی سنجیده شود. ویژگی‌های آزمایش خوب:
\begin{itemize}
  \item روی افراد متنوع انجام شود.
  \item فقط پاسخ درست را نسنجد؛ خطاها را هم توضیح دهد.
  \item به یک سن، یک وضعیت یا یک گروه بسیار خاص محدود نباشد.
  \item اختلالات و تفاوت‌های شناختی را هم در نظر بگیرد.
  \item در شرایط و آزمایش‌های مختلف قابل تکرار باشد.
  \item بتواند پیش‌بینی‌های جدید تولید کند.
\end{itemize}

برای نمونه، اگر کسی به‌جای «برنج» کلمه‌ای دیگر بشنود، نظریه باید بتواند توضیح دهد این خطا از کجا آمده است: نویز صوتی، \eng{context}، انتظار قبلی، یا محدودیت توجه.

\chapter{معماری شناختی و حافظه}

\section{چرا معماری شناختی لازم است؟}
اگر بخواهیم چند نظریهٔ شناختی را در یک سیستم هوش مصنوعی وارد کنیم، باید بدانیم جریان اطلاعات چگونه میان آن‌ها حرکت می‌کند. ورودی کجا می‌رود؟ حافظه کجا وارد می‌شود؟ کدام ماژول تصمیم می‌گیرد؟ عمل چگونه تولید می‌شود؟ پاسخ این پرسش‌ها همان \term{معماری شناختی}{Cognitive Architecture} است.

\begin{definitionbox}{تعریف معماری شناختی}
معماری شناختی یعنی مدل‌کردن فرایندهای شناختی انسان در قالب یک سیستم یکپارچه که ماژول‌هایی مانند ادراک، حافظه، بازیابی، تصمیم‌گیری و عمل در آن با هم تعامل دارند.
\end{definitionbox}

یک معماری شناختی باید تا حدی با مغز قابل نگاشت باشد؛ مثلاً اگر ماژول بینایی دارد، باید بتوان گفت با نواحی بینایی مثل \eng{occipital lobe} متناظر است. اگر حافظهٔ اعلانی دارد، باید نسبت آن با \eng{hippocampus} توضیح داده شود.

\section{مثال جارو رباتیک}
فرض کنید ربات جاروبرقی با یک فنجان پر از آب روبه‌رو می‌شود. یک مدل ساده ممکن است فقط مانع را کنار بزند، اما انسان چنین کاری نمی‌کند؛ چون می‌داند فنجان ممکن است بشکند، آب ممکن است بریزد و زمین خیس شود.

برای رفتار انسانی‌تر، ربات به چیزهایی بیشتر از تشخیص شیء نیاز دارد:
\begin{itemize}
  \item \term{دانش عامه}{common-sense knowledge}
  \item حافظهٔ کوتاه‌مدت برای نگه‌داشتن وضعیت فعلی
  \item حافظهٔ اعلانی برای دانستن \eng{facts}
  \item حافظهٔ رویه‌ای برای اجرای مهارت‌ها
  \item انتخاب عمل و بازخوردگیری از عمل قبلی
\end{itemize}

\section{انواع حافظه}
در کلاس، حافظه ابتدا به دو دستهٔ اصلی تقسیم شد:

\subsection{حافظهٔ کوتاه‌مدت و حافظهٔ کاری}
\term{حافظهٔ کوتاه‌مدت}{Short-Term Memory} یا \term{حافظهٔ کاری}{Working Memory} یک فضای محدود و موقت برای نگهداری و پردازش اطلاعات فعلی است. مثال ساده، نگه داشتن یک شماره تلفن برای چند ثانیه است.

\subsection{حافظهٔ بلندمدت}
\term{حافظهٔ بلندمدت}{Long-Term Memory} پایدارتر است و خود به دو دستهٔ مهم تقسیم می‌شود:

\begin{itemize}
  \item \term{حافظهٔ اعلانی}{Declarative Memory}: \eng{facts} و دانش صریح؛ مثل اینکه آب زمین را خیس می‌کند.
  \item \term{حافظهٔ رویه‌ای}{Procedural Memory}: مهارت‌ها و \eng{procedure}ها؛ مثل رانندگی، آشپزی یا حرکت دادن بازوی ربات.
\end{itemize}

\subsection{نسبت \eng{Chain of Thought} با حافظه}
در کلاس تأکید شد که \eng{Chain of Thought} را نباید مستقیماً معادل حافظهٔ رویه‌ای دانست. \eng{CoT} بیشتر به آشکار کردن یا هدایت گام‌های استدلال کمک می‌کند و به بازیابی دانش اعلانی و \eng{reasoning} نزدیک‌تر است. ممکن است اجرای یک روش مرحله‌به‌مرحله به استفاده از مهارت‌های رویه‌ای هم کمک کند، اما خود آن یک مدل کامل از حافظهٔ رویه‌ای انسان نیست.

\section{اختلالات و اعتبارسنجی تقسیم‌بندی حافظه}
یک نظریهٔ شناختی وقتی قوی‌تر می‌شود که بتواند اختلالات را نیز توضیح دهد. اگر نظریه می‌گوید حافظه به کوتاه‌مدت، بلندمدت، اعلانی و رویه‌ای تقسیم می‌شود، باید بتواند نشان دهد کدام اختلال با کدام بخش سازگارتر است.

برای مثال، استرس می‌تواند بازیابی حافظهٔ اعلانی را مختل کند: فرد چیزی را می‌داند، اما در لحظه نمی‌تواند آن را توضیح دهد یا به یاد آورد. اگر استرس روی خود عمل گفتار، لکنت یا اجرای مهارت اثر بگذارد، می‌توان بخشی از آن را به حافظهٔ رویه‌ای نیز ربط داد. در پارکینسون، به‌ویژه در مراحل اولیه، اختلال‌های حرکتی و مهارت‌های رویه‌ای برجسته‌ترند و با مدارهایی مثل \eng{basal ganglia} سازگارند؛ هرچند در مراحل بعدی ممکن است کارکردهای شناختی دیگر نیز درگیر شوند.

\section{\eng{ACT} و \eng{ACT-R}}
معماری \eng{ACT} از کارهای \eng{John R. Anderson} در دههٔ ۱۹۷۰ آغاز شد. نسخه‌های بعدی، به‌ویژه \eng{ACT-R}، تلاش می‌کنند ساختار حافظه، تولید عمل و نگاشت به مغز را دقیق‌تر مدل کنند.

در این معماری، حافظهٔ اعلانی فقط یک مخزن داده نیست؛ می‌تواند به توضیح‌پذیری هم کمک کند. اگر ربات مسیرش را عوض کند، دانش اعلانی می‌تواند توضیحی از جنس «فنجان پر از آب شکستنی و در مسیر بود» فراهم کند. حافظهٔ رویه‌ای نیز معمولاً به شکل قاعده‌های \eng{if-then} یا \eng{production rule} فهمیده می‌شود؛ مثلاً «اگر مانع شکستنی و پرخطر در مسیر بود، سرعت را کم کن یا مسیر را تغییر بده».

\section{انتقال از حافظهٔ کوتاه‌مدت به بلندمدت}
اطلاعات معمولاً با تمرین، تکرار و استفاده از حافظهٔ کوتاه‌مدت به حافظهٔ بلندمدت منتقل می‌شوند، اما کلاس روی چند عامل شتاب‌دهنده نیز تأکید کرد:
\begin{itemize}
  \item احساس شدید، مثل حادثه، ترس، شادی بزرگ یا تولد فرزند.
  \item تعجب و غیرمنتظره بودن؛ به همین دلیل تجربهٔ آموزشی متفاوت می‌تواند ماندگارتر شود.
  \item تمایز و برجستگی؛ چیزی که از زمینهٔ معمول جدا باشد راحت‌تر به خاطر می‌ماند.
  \item علامت‌گذاری، تصویرسازی ذهنی، ساخت مثال و پیدا کردن ارتباط بین مفاهیم.
\end{itemize}

از منظر آموزشی، پرسش مهم این است که چگونه می‌توان یادگیری را طوری طراحی کرد که اطلاعات مهم سریع‌تر و پایدارتر وارد حافظهٔ بلندمدت شوند.

این نگاشت‌ها دقیق و جدا از هم نیستند. مغز شبکه‌ای و تعاملی عمل می‌کند، اما همین نگاشت تقریبی برای اعتبارسنجی معماری شناختی مهم است.

\section{اعتبارسنجی معماری شناختی}
اگر یک معماری ادعا کند شبیه انسان پردازش می‌کند، باید فقط خروجی آن با انسان مقایسه نشود. باید \eng{sequence} فعال‌شدن ماژول‌ها نیز با دادهٔ انسانی مقایسه شود. مثلاً در یک \eng{task} بینایی، می‌توان از \eng{fMRI}، \eng{EEG}، زمان واکنش یا \eng{eye-tracking} استفاده کرد و بررسی کرد آیا جریان فعالیت انسان با جریان ماژول‌های مدل سازگار است یا نه.

\begin{assignmentbox}{فاز اول پروژه: پیدا کردن \eng{dataset} شناختی}
دانشجویان باید \eng{dataset}ی پیدا کنند که برای مدل‌سازی معماری شناختی مناسب باشد. \eng{dataset} خوب بهتر است شامل ورودی، خروجی یا عمل انسان، و در صورت امکان \eng{sequence}، زمان واکنش، \eng{fMRI}، \eng{EEG}، مسیر نگاه یا گزارش فرایند حل مسئله باشد. خروجی مورد انتظار یک سند کوتاه بود: معرفی \eng{dataset}، ورودی انسان، خروجی یا رفتار انسان، وجود یا نبود سیگنال مغزی/رفتاری/زمانی، امکان دادن همان ورودی به مدل، \eng{task} قابل انجام و روش مقایسهٔ انسان و مدل. تمرین می‌توانست دو نفره باشد و تحویل آن از طریق \eng{LMS} انجام شود. در رونوشت جلسه، تاریخ دقیق ددلاین کمی نامطمئن بود و به حوالی جلسهٔ یکشنبه اشاره شده بود.
\end{assignmentbox}

\chapter{حافظهٔ کاری، بار شناختی و مدل‌های زبانی}

\section{\eng{Cognitive Load Theory}}
\term{نظریهٔ بار شناختی}{Cognitive Load Theory} بر محدودیت حافظهٔ کاری تأکید دارد. انسان نمی‌تواند در هر لحظه همهٔ اطلاعات محیط یا همهٔ شاخه‌های یک مسئله را پردازش کند. بنابراین باید اطلاعات را انتخاب، سازمان‌دهی و فشرده کند.

\section{عدد \eng{7 ± 2} و مفهوم \eng{Chunk}}
مقالهٔ معروف \eng{The Magical Number Seven, Plus or Minus Two} از \eng{George Miller} در سال ۱۹۵۶ نشان داد که ظرفیت پردازش فوری انسان محدود است. اما واحد حافظه لزوماً یک حرف یا عدد منفرد نیست؛ واحد مهم \term{chunk}{Chunk} است.

اگر عددهای \eng{1 3 6 9} را جدا ببینیم، چهار آیتم‌اند. اما اگر آن را سال \eng{1369} بدانیم، کل آن می‌تواند یک \eng{chunk} شود. بنابراین \eng{chunking} به حافظهٔ بلندمدت وابسته است: دانش قبلی کمک می‌کند اطلاعات جدید را فشرده‌تر نگه داریم.

\begin{correctionbox}{دقت مفهومی}
در ادبیات جدیدتر، عدد \eng{7 ± 2} قانون ثابت همهٔ انسان‌ها نیست. پیام اصلی این است که حافظهٔ کاری محدود است و عملکرد به نوع اطلاعات، \eng{chunking}، تمرین، تخصص و شرایط فرد بستگی دارد.
\end{correctionbox}

\section{حافظهٔ کوتاه‌مدت و بلندمدت در \eng{LLM}}
در مدل‌های زبانی بزرگ:
\begin{itemize}
  \item حافظهٔ بلندمدت را می‌توان تقریباً معادل وزن‌ها و پارامترهای آموزش‌دیده دانست.
  \item حافظهٔ کوتاه‌مدت یا کاری را می‌توان تقریباً معادل \eng{prompt} و \eng{context window} دانست.
  \item \eng{RAG} می‌تواند مانند حافظهٔ بلندمدت خارجی عمل کند.
\end{itemize}

این تشبیه کامل نیست، اما برای تحلیل محدودیت‌های مدل مفید است. وقتی \eng{prompt} طولانی، پرنویز یا متناقض شود، مدل ممکن است مانند انسانی با بار شناختی بالا دچار افت عملکرد شود.

\section{\eng{Multi-Hop Reasoning}}
در \term{استدلال چندمرحله‌ای}{Multi-Hop Reasoning} پاسخ نهایی به چند قطعه اطلاعات و چند گام استدلال وابسته است. مدل باید اطلاعات را نگه دارد، ارتباط دهد، نتیجهٔ میانی بسازد و از آن برای گام بعدی استفاده کند. به همین دلیل این \eng{task} برای سنجش حافظهٔ کاری مدل مناسب است.

\section{دو نوع فشار حافظه‌ای در \eng{LLM}}
\subsection{\eng{Context Saturation}}
در \eng{Context Saturation} اطلاعات مهم در میان حجم زیادی از نویز پخش می‌شود. با افزایش نویز، مدل ممکن است نتواند دادهٔ مرتبط را پیدا کند و \eng{success rate} کاهش یابد.

\subsection{\eng{Attentional Residue}}
در \eng{Attentional Residue} اثر اطلاعات یا \eng{task}های قبلی روی \eng{task} فعلی باقی می‌ماند. مدل ممکن است سؤال جدید را با زمینهٔ قبلی قاطی کند. در تجربهٔ عملی، گاهی \eng{reset} کردن مکالمه یا بازنویسی \eng{prompt} همین مشکل را کم می‌کند.

\subsection{طراحی آزمایش برای محدودیت حافظهٔ کاری}
برای نشان دادن محدودیت حافظهٔ کاری مدل، فقط طولانی کردن \eng{prompt} کافی نیست. در کلاس چند طراحی آزمایشی مطرح شد: اضافه کردن نویز بین اطلاعات مرتبط، پخش کردن شواهد لازم در بخش‌های دور از هم، ارجاع به بخش‌های قبلی مکالمه، تغییر خواسته پس از ارائهٔ یک دستور یا کد، و قرار دادن یک \eng{task} مزاحم قبل از \eng{task} اصلی. اگر مدل با افزایش نویز یا تغییر ترتیب شواهد افت کند، می‌توان گفت مشکل فقط اندازهٔ \eng{context window} نیست؛ مدیریت توجه و بازیابی اطلاعات نیز محدود است.

\subsection{تکامل آزمون‌های هوش مصنوعی}
در ابتدا آزمون‌هایی مثل \eng{Turing Test} یا معیارهای کلاسیک \eng{accuracy}، \eng{precision} و \eng{recall} کافی به نظر می‌رسیدند. سپس مسئلهٔ \term{تعمیم خارج از توزیع}{Out-of-Distribution Generalization} مهم شد. اکنون یکی از مسیرهای جدی، آزمون‌های شناختی است: آیا مدل مانند انسان محدودیت حافظهٔ کاری دارد؟ آیا با بار شناختی، نویز، جابه‌جایی توجه یا قیاس دور دچار خطای مشابه می‌شود؟

\section{راهکار شناختی: \eng{Chunking} و \eng{Divide and Conquer}}
یک راه کاهش بار شناختی، تقسیم مسئله به زیرمسئله‌هاست. در \eng{agent}های \eng{Long-Horizon}، هر \eng{sub-goal} می‌تواند مثل یک \eng{memory chunk} عمل کند. پس از انجام \eng{sub-goal}، لازم نیست همهٔ جزئیات آن در حافظه بماند؛ فقط نتیجهٔ مهم نگه داشته می‌شود.

مثلاً در \eng{Blocks World}:
\begin{enumerate}
  \item بلوک آبی clear شود.
  \item بلوک نارنجی clear شود.
  \item بلوک آبی روی بلوک نارنجی قرار بگیرد.
\end{enumerate}

فرمول کلی \eng{divide and conquer}:
\[
T(n)=aT(n/b)+f(n)
\]
در برخی مسائل، تقسیم می‌تواند هزینه را از مرتبه‌های سنگین‌تر به حدود \(O(n\log n)\) کاهش دهد، اما این یک نتیجهٔ کلی برای همهٔ \eng{task}ها نیست. اگر مسئله ساختار تقسیم‌پذیر نداشته باشد، \eng{chunking} لزوماً کمک نمی‌کند؛ مانند برخی نسخه‌های سادهٔ \eng{Gripper Robot}.

تفاوت مهم انسان و مدل این است که انسان اغلب خودش تشخیص می‌دهد چه زمانی \eng{chunk} بسازد، چه چیزی را فراموش کند و چه نتیجه‌ای را نگه دارد. در \eng{agent}های فعلی، معمولاً طراح باید \eng{sub-goal}ها، واحدهای حافظه‌ای و سیاست فراموشی را مشخص کند. بنابراین ایدهٔ شناختی مفید است، اما هنوز نیاز به طراحی محاسباتی دقیق دارد.

\chapter{ترنسفورمر، توجه و حافظهٔ کاری}

\section{حافظه در \eng{Transformer}}
در یک \eng{Transformer}:
\begin{itemize}
  \item دانش بلندمدت در وزن‌هاست.
  \item پردازش ورودی فعلی در لایه‌های \eng{attention} و \eng{feed-forward} رخ می‌دهد.
  \item \eng{Self-Attention} هر \eng{token} را با \eng{token}های دیگر مقایسه می‌کند و \eng{embedding} آن را \eng{contextual} می‌سازد.
\end{itemize}

در کلاس ابتدا سه خانوادهٔ کلی \eng{Transformer} مرور شد: \eng{encoder-only}، \eng{decoder-only} و \eng{encoder-decoder}. برای بحث حافظهٔ کاری، تمرکز بیشتر روی این ایده بود که مدل در زمان \eng{inference} با ورودی فعلی چه می‌کند. وزن‌های مدل مثل حافظهٔ بلندمدت‌اند؛ اما \eng{prompt} یا ورودی جاری، چیزی است که باید در لحظه پردازش شود. به همین دلیل \eng{attention layer} یکی از جاهای مهم برای تحلیل حافظهٔ کاری مدل است.

\section{\eng{Contextual Embedding}}
کلمهٔ «شیر» در «شیر جنگل»، «شیر خوردنی» و «شیر آب» یک \eng{embedding} ثابت نباید داشته باشد. \eng{Transformer} با \eng{attention}، \eng{embedding} اولیه را به \eng{embedding} وابسته به \eng{context} تبدیل می‌کند. همین کار برای مسائلی مثل \term{رفع ابهام معنای واژه}{Word Sense Disambiguation} و \term{حل ارجاع}{Coreference Resolution} ضروری است.

در ابتدا هر \eng{token} یک بازنمایی اولیه دارد که از وزن‌های آموزش‌دیده می‌آید. اما این بازنمایی هنوز نمی‌داند کلمه در جملهٔ فعلی چه نقشی دارد. مثلاً در «علی رفت مدرسه. او کتابش را برداشت.» مدل باید بفهمد «او» به «علی» برمی‌گردد. بنابراین \eng{embedding} «او» باید با توجه به \eng{context} به بازنمایی «علی» نزدیک‌تر شود. این تغییر همان کاری است که حافظهٔ کاری مدل روی ورودی فعلی انجام می‌دهد.

\section{\eng{Query, Key, Value}}
برای هر \eng{token} سه بردار ساخته می‌شود:
\[
Q=xW_Q,\qquad K=xW_K,\qquad V=xW_V
\]
سپس وزن توجه به‌صورت ساده از ضرب \(QK^\top\) و نرمال‌سازی به دست می‌آید و روی \(V\) اعمال می‌شود:
\[
\operatorname{Attention}(Q,K,V)=\operatorname{softmax}\left(\frac{QK^\top}{\sqrt{d_k}}\right)V
\]

تعبیر شهودی این است: \eng{Query} می‌پرسد «من برای فهم خودم دنبال چه اطلاعاتی هستم؟»، \eng{Key} می‌گوید «هر \eng{token} چه نوع اطلاعاتی برای ارائه دارد؟» و \eng{Value} همان محتوایی است که پس از تعیین وزن‌ها وارد بازنمایی جدید می‌شود. اگر \eng{Query} یک \eng{token} با \eng{Key} \eng{token} دیگری سازگار باشد، وزن توجه بین آن‌ها بیشتر می‌شود. نتیجه این است که هر \eng{token} فقط خودش نیست؛ نسخهٔ جدید آن شامل اطلاعاتی از \eng{token}های مرتبط در همان \eng{context} است.

\section{آیا کافی است \eng{attention} را بزرگ‌تر کنیم؟}
هدف همیشه افزایش طول \eng{context} یا بزرگ‌تر کردن \eng{attention} نیست. انسان هم فقط با بزرگ‌تر شدن حافظه بهتر نمی‌شود؛ بلکه با راهبردهایی مثل \eng{chunking}، انتخاب و سازمان‌دهی بهتر عمل می‌کند. در مدل‌ها نیز مسئلهٔ اصلی این است که اطلاعات مرتبط بهتر بازیابی شود و اطلاعات نامرتبط کمتر مزاحمت ایجاد کند.

یک ایدهٔ ساده این است که \eng{attention score}های کوچک‌تر از یک آستانه حذف شوند، اما این کار همیشه درست نیست. \eng{attention} نسبی است: یک ارتباط ممکن است نسبت به یک \eng{token} ضعیف باشد، اما در زنجیرهٔ روابط نقش غیرمستقیم مهمی داشته باشد. برای مثال اگر \(A\) به \(B\) و \(B\) به \(C\) مرتبط باشد، حذف سادهٔ ارتباط‌های ظاهراً کم‌وزن می‌تواند رابطهٔ دوربرد \(A\) و \(C\) را خراب کند.

\section{\eng{Local Attention}}
\eng{Local Attention} محاسبهٔ \eng{attention} را به پنجره‌های محلی محدود می‌کند. مزیت آن کاهش هزینهٔ محاسباتی و فشار حافظه‌ای است. خطر آن از دست رفتن وابستگی‌های دوربرد است. بنابراین معمولاً باید راهی برای ترکیب اطلاعات محلی در لایه‌های بالاتر وجود داشته باشد.

\eng{Local Attention} به معنی حذف کامل بخش‌های متن نیست؛ بیشتر شبیه نگاه مرحله‌ای است. مدل در هر پنجره یک \eng{representation} محلی می‌سازد و سپس باید این بازنمایی‌های محلی را در لایه یا مرحله‌ای بالاتر ترکیب کند. این ایده به \eng{divide and conquer} شباهت دارد، اما فقط وقتی مفید است که ترکیب سطح بالاتر اطلاعات مهم دوربرد را دوباره در دسترس قرار دهد.

\section{\eng{Multi-Head Attention} و \eng{Mixture of Experts}}
\eng{Multi-Head Attention} را می‌توان نگاه از چند زاویه دانست: یک \eng{head} ممکن است روابط نحوی را ببیند، دیگری روابط معنایی یا ارجاعی را. \eng{Mixture of Experts} نیز با فعال کردن متخصص‌های متفاوت، ظرفیت مدل را بهتر استفاده می‌کند. این‌ها مستقیماً معادل حافظهٔ کاری نیستند، اما می‌توانند پردازش \eng{context} را غنی‌تر کنند.

بنابراین در این بخش باید بین دو ادعا فرق گذاشت: این سازوکارها می‌توانند به استفادهٔ بهتر از اطلاعات کمک کنند، اما خودشان پاسخ کامل به مسئلهٔ محدودیت حافظهٔ کاری نیستند. اگر مدل نداند چه اطلاعاتی را نگه دارد، چه چیزی را خلاصه کند و چه چیزی را کنار بگذارد، صرف داشتن \eng{head}های بیشتر یا متخصص‌های بیشتر لزوماً مشکل بار شناختی را حل نمی‌کند.

\chapter{سوگیری شناختی و سه نظریهٔ توضیح‌دهنده}

\section{تعریف \eng{Cognitive Bias}}
\term{سوگیری شناختی}{Cognitive Bias} یعنی سوگیری در فرایند تفکر، استدلال یا تصمیم‌گیری که باعث می‌شود فرد به نتیجه‌ای غلط یا غیربهینه برسد. فرد معمولاً برای تصمیم خود دلیل می‌آورد، اما خود آن دلیل می‌تواند سوگیرانه باشد.

\begin{assignmentbox}{نکات اجرایی جلسهٔ سوگیری}
به‌دلیل عقب‌افتادن بعضی جلسات و محدودیت دسترسی به منابع آنلاین، استاد گفت ممکن است بخشی از جبران کلاس‌ها به‌صورت آنلاین یا آفلاین انجام شود، اما تأکید کرد این درس به بحث کلاسی نیاز دارد. همچنین اشاره شد پروژهٔ اولیه به‌دلیل محدودیت اینترنت ممکن است تغییر کند یا بخشی از نمره با کار مطالعاتی/پروژهٔ جایگزین پوشش داده شود.
\end{assignmentbox}

\section{انتخاب خط پژوهشی}
در ابتدای جلسهٔ سوگیری، استاد بحثی دربارهٔ انتخاب \eng{line of research} در تحصیلات تکمیلی مطرح کرد. نکتهٔ اصلی این بود که دانشجو باید آگاهانه نسبت خود را با حوزه‌های حساس، کاربردی، درمانی، صنعتی یا دفاعی مشخص کند و فقط از روی محافظه‌کاری مسیر پژوهش را انتخاب نکند. این بخش محتوای فنی مستقیم نداشت، اما در چارچوب درس مهم بود چون پروژهٔ پایانی باید به یک مسیر پژوهشی قابل دفاع و ادامه‌دار وصل شود.

\section{نمونه‌ها}
\subsection{\eng{Confirmation Bias}}
فرد باور قبلی دارد و اطلاعات جدید را طوری انتخاب یا تفسیر می‌کند که همان باور قبلی تأیید شود.

مثال کلاس این بود که فرد از گذشته یک باور قوی دارد و هنگام خواندن خبر، کتاب یا دادهٔ جدید، بیشتر به شواهدی توجه می‌کند که همان باور را تقویت کنند. مشکل فقط «ندیدن» داده نیست؛ گاهی داده دیده می‌شود، اما طوری تفسیر می‌شود که مدل ذهنی قبلی تغییر نکند.

\subsection{\eng{Anchoring Bias}}
اولین داده یا برداشت اولیه وزن زیادی پیدا می‌کند و روی قضاوت‌های بعدی اثر می‌گذارد.

به همین دلیل «اولین‌ها» در تصمیم‌گیری وزن زیادی دارند: اولین برداشت از یک فرد، اولین عددی که در مذاکره شنیده می‌شود، یا اولین توضیحی که دربارهٔ یک مسئله دریافت می‌کنیم. این وزن زیاد گاهی مفید است، چون تصمیم سریع را ممکن می‌کند، اما در مسائل پیچیده می‌تواند قضاوت‌های بعدی را به سمت همان نقطهٔ اولیه بکشد.

\subsection{\eng{Halo Effect}}
یک ویژگی مثبت فرد به ویژگی‌های دیگر او تعمیم داده می‌شود؛ مثلاً ظاهر مرتب باعث می‌شود ناخودآگاه او را باهوش‌تر یا قابل‌اعتمادتر فرض کنیم.

پدیدهٔ \eng{celebrity} نیز تا حدی با همین سازوکار توضیح داده می‌شود: یک ویژگی برجسته یا جذاب باعث می‌شود مخاطب اعتبار فرد را به حوزه‌هایی تعمیم دهد که لزوماً در آن‌ها تخصصی ندارد.

\section{آیا \eng{bias} همیشه بد است؟}
نه. بسیاری از \eng{bias}ها در تکامل باقی مانده‌اند چون تصمیم‌گیری سریع و کم‌هزینه را ممکن می‌کنند. مشکل وقتی رخ می‌دهد که تصمیم مهم، پیچیده یا پرپیامد باشد و پاسخ سریع و سوگیرانه جای تحلیل دقیق را بگیرد.

\section{توضیح اول: \eng{Cognitive Load Theory}}
اگر ذهن نمی‌تواند همهٔ \eng{decision tree} را \eng{load} کند، ناچار \eng{greedy} عمل می‌کند: بخشی از اطلاعات را انتخاب می‌کند و با همان تصمیم می‌گیرد. این تصمیم ممکن است \eng{feasible} باشد، اما \eng{optimal} نباشد. راهکارهای کاهش \eng{bias} از این منظر:
\begin{itemize}
  \item ساختارمند کردن اطلاعات
  \item \eng{chunking}
  \item چک‌لیست
  \item تقسیم مسئله به بخش‌های کوچک‌تر
  \item مدیریت \eng{attention}
\end{itemize}

این توضیح می‌گوید سوگیری الزاماً از بی‌منطقی کامل نمی‌آید؛ گاهی از محدودیت منابع پردازشی می‌آید. وقتی مسئله بزرگ است، فرد نمی‌تواند همهٔ شاخه‌های ممکن را هم‌زمان نگه دارد، پس به یک راه‌حل محلی و کم‌هزینه تکیه می‌کند. این راه‌حل در شرایط روزمره ممکن است کافی باشد، اما در تصمیم‌های پرهزینه یا حساس باید ساختار مسئله را صریح‌تر کرد تا فشار حافظهٔ کاری کمتر شود.

\section{توضیح دوم: \eng{Dual Process Theory}}
طبق نظریهٔ دوفرایندی، انسان دو سبک پردازش دارد:
\begin{itemize}
  \item \term{سیستم ۱}{System 1}: سریع، شهودی، کم‌هزینه و مناسب واکنش فوری.
  \item \term{سیستم ۲}{System 2}: کندتر، تحلیلی، مرحله‌به‌مرحله و پرهزینه‌تر.
\end{itemize}

سوگیری معمولاً وقتی رخ می‌دهد که مسئله به سیستم ۲ نیاز دارد اما سیستم ۱ پاسخ سریع می‌دهد و همان پاسخ اصلاح نمی‌شود.

سیستم ۱ همیشه بد نیست. در بسیاری از موقعیت‌های روزمره، واکنش سریع و شهودی لازم است و اگر قرار باشد برای هر تصمیم کوچک تحلیل عمیق انجام دهیم، هزینهٔ زمانی و شناختی بیش از حد می‌شود. خطر وقتی بالا می‌رود که مسئله مهم، پیچیده، تازه یا پرپیامد است، اما ذهن همچنان با پاسخ سریع پیش‌فرض جلو می‌رود.

برای کاهش سوگیری از این منظر، باید عمداً شرایط فعال‌شدن سیستم ۲ را فراهم کرد: مکث قبل از تصمیم، نهایی نکردن اولین پاسخ، تصمیم نگرفتن در خستگی یا فشار روانی، جمع کردن شواهد بیشتر، صریح کردن معیارهای تصمیم و نقد پاسخ اولیه.

\section{رفتار شبیه سیستم ۲ در \eng{LLM}}
در \eng{LLM} واقعاً سیستم ۱ و سیستم ۲ انسانی وجود ندارد، اما می‌توان خروجی مدل را به رفتار سریع یا تحلیلی نزدیک کرد. روش‌ها:
\begin{itemize}
  \item درخواست \eng{reasoning} مرحله‌به‌مرحله
  \item \eng{self-critique}
  \item \eng{few-shot prompting} با مثال‌های مرزی
  \item \eng{Chain of Thought}
  \item \eng{Tree of Thought}
  \item نقد پاسخ اولیه و بازنویسی پاسخ
\end{itemize}

برای نمونه، \eng{self-critique} را می‌توان چنین دید: مدل ابتدا یک پاسخ اولیه می‌دهد، سپس از خودش می‌خواهد همان پاسخ را از نظر فرض‌ها، خطاها و شواهد بررسی کند، و بعد نسخهٔ اصلاح‌شده را تولید کند. در \eng{Tree of Thought} نیز مدل فقط یک مسیر استدلال را دنبال نمی‌کند؛ چند شاخهٔ ممکن را بررسی می‌کند و سپس مسیر بهتر را انتخاب می‌کند. این روش‌ها دقیقاً سیستم ۲ انسانی نیستند، اما خروجی را از پاسخ سریع و تک‌مسیره دور می‌کنند.

\section{توضیح سوم: \eng{Predictive Coding}}
طبق \term{کدگذاری پیش‌بینانه}{Predictive Coding}، مغز فقط گیرندهٔ منفعل داده نیست؛ بلکه دائماً از بالا به پایین پیش‌بینی تولید می‌کند و ورودی حسی پایین‌به‌بالا را با آن مقایسه می‌کند.

\[
\text{\lr{Prediction Error}} = \text{\lr{Input}} - \text{\lr{Prediction}}
\]

اگر خطا کوچک باشد، سیستم پایدار می‌ماند. اگر خطا بزرگ باشد، یا \eng{prior} باید اصلاح شود یا \eng{input} جدید جمع‌آوری شود.

\chapter{\eng{Predictive Coding}، بیز و مدل‌های عصبی}

\section{\eng{Prior}، \eng{Likelihood} و \eng{Posterior}}
در زبان بیزی:
\[
\text{\lr{Posterior}} \propto \text{\lr{Prior}} \times \text{\lr{Likelihood}}
\]

در این نگاشت:
\begin{itemize}
  \item حافظهٔ بلندمدت و باور قبلی شبیه \eng{Prior} است.
  \item ورودی فعلی شبیه \eng{Likelihood} است.
  \item باور به‌روزشده شبیه \eng{Posterior} است.
\end{itemize}

مثال کارآگاه و صحنهٔ جرم در کلاس برای همین نگاشت بود. کارآگاه با \eng{prior} وارد صحنه می‌شود، اما شواهد صحنه نقش \eng{likelihood} دارند. اگر شواهد معتبر و قوی باشند، باور اولیه باید جابه‌جا شود. اگر شواهد ضعیف، مبهم یا کم‌اعتبار باشند، باور قبلی وزن بیشتری پیدا می‌کند.

\section{Precision}
\term{Precision}{Precision} یعنی میزان اطمینان یا وزن یک منبع اطلاعاتی. اگر \eng{prior} بر اساس تجربهٔ طولانی و شواهد زیاد شکل گرفته باشد، \eng{precision} آن بالاست. اگر \eng{input} واضح، معتبر و مستقیم باشد، \eng{precision input} بالاست.

فرمول میانگین \eng{posterior}:
\[
\mu_{\text{posterior}}=
\frac{\pi_p\mu_p+\pi_i\mu_i}{\pi_p+\pi_i}
\]
که در آن \(\pi_p\) \eng{precision prior} و \(\pi_i\) \eng{precision input} است.

بازنویسی فرمول:
\[
\mu_{\text{posterior}}
=
\mu_p+
\frac{\pi_i}{\pi_p+\pi_i}(\mu_i-\mu_p)
\]

عبارت \((\mu_i-\mu_p)\) همان \eng{prediction error} است و ضریب قبل از آن وزن خطاست.

\section{چه زمانی \eng{belief update} نمی‌شود؟}
\begin{itemize}
  \item وقتی \eng{prediction error} صفر باشد.
  \item وقتی \eng{precision input} بسیار کم باشد.
  \item وقتی \eng{precision prior} بسیار بیشتر از \eng{precision input} باشد.
\end{itemize}

حالت سوم توضیح خوبی برای \eng{confirmation bias} است: \eng{prior} آن‌قدر قوی است که شواهد جدید نمی‌توانند آن را جابه‌جا کنند.

مثال رسانه و تخصص نیز همین را نشان می‌دهد. فردی که دربارهٔ یک موضوع تخصص و تجربهٔ بلندمدت دارد، \eng{precision prior} بالایی دارد و با هر خبر ضعیفی نظرش عوض نمی‌شود. این همیشه بد نیست. مشکل وقتی رخ می‌دهد که \eng{prior} قوی ولی غلط باشد، یا فرد عمداً \eng{precision input} شواهد مخالف را نزدیک صفر فرض کند.

\section{وقتی خطا وجود دارد، سیستم چه می‌کند؟}
اگر \eng{prediction error} معنی‌دار باشد، دو مسیر کلی داریم:
\begin{itemize}
  \item اصلاح \eng{prior}: باور قبلی عوض می‌شود تا با شواهد سازگارتر شود.
  \item جمع‌آوری یا تغییر \eng{input}: فرد دادهٔ جدید می‌گیرد، زاویهٔ نگاه را عوض می‌کند یا محیط را بهتر اندازه‌گیری می‌کند.
\end{itemize}

اگر هیچ‌کدام انجام نشود، سیستم ممکن است خطا را با توجیه، نادیده‌گرفتن یا بازتفسیر شواهد حل کند. این همان نقطه‌ای است که \eng{predictive coding} می‌تواند به توضیح سوگیری شناختی کمک کند.

\section{کاهش bias در \eng{LLM} با این چارچوب}
اگر وزن‌های مدل را \eng{prior} بدانیم، برای کاهش تکیهٔ بیش‌ازحد بر \eng{prior} عمومی مدل می‌توان \eng{precision input} را بالا برد یا \eng{precision prior} نامطلوب را کم کرد:
\begin{itemize}
  \item \eng{RAG}: دادن شواهد بیرونی معتبر و مرتبط.
  \item \eng{Few-shot prompting}: روشن کردن الگوی مطلوب با نمونه‌های مشخص.
  \item \eng{Negative prompting}: محدود کردن فرض‌ها یا سوگیری‌های ناخواسته.
  \item \eng{Role playing}: هدایت مدل به نقش دقیق‌تر، مثل «داور بی‌طرف».
\end{itemize}

\section{تفاوت با شبکه‌های عصبی رایج}
در شبکهٔ عصبی رایج، معمولاً خطا در خروجی نهایی محاسبه می‌شود و با \eng{Backpropagation} به کل شبکه برمی‌گردد. این خطا \eng{global} و \eng{end-to-end} است. در \eng{Predictive Coding}، ایده این است که خطا به‌صورت \eng{local} و پیوسته در سطوح مختلف محاسبه شود.

تفاوت دوم جریان اطلاعات است:
\begin{itemize}
  \item شبکه‌های رایج عمدتاً \eng{bottom-up} هستند.
  \item \eng{predictive coding} هم \eng{bottom-up} و هم \eng{top-down} دارد.
\end{itemize}

\section{آیا \eng{GAN} شبیه \eng{Predictive Coding} است؟}
شباهت استعاری وجود دارد: در \eng{GAN} یک \eng{generator} تولید می‌کند و \eng{discriminator} ارزیابی می‌کند؛ در \eng{predictive coding} نیز سیستم پیش‌بینی تولید می‌کند و با \eng{input} مقایسه می‌کند. اما نگاشت دقیق نیست. در \eng{predictive coding}، خطای محلی و جریان پیوستهٔ \eng{top-down/bottom-up} نقش مرکزی دارد؛ در \eng{GAN} خطا و یادگیری در چارچوب \eng{adversarial} و معمولاً با \eng{backpropagation} انجام می‌شود.

\section{پردازنده‌های \eng{Neuromorphic}}
در اکوسیستم رایج هوش مصنوعی، بیشتر مدل‌ها روی \eng{GPU} و با \eng{backpropagation} آموزش یا اجرا می‌شوند. در مقابل، پردازنده‌های \term{نورومورفیک}{Neuromorphic} می‌کوشند از ایده‌های مغز الهام بگیرند: نورون، \eng{spike}، سیناپس، پردازش محلی و گاهی یادگیری رویدادمحور. این سخت‌افزارها هنوز جایگزین عمومی \eng{GPU} نشده‌اند، چون اکوسیستم نرم‌افزاری و شتاب‌دهنده‌های فعلی حول \eng{GPU} شکل گرفته است. برای استفادهٔ وسیع، مزیت عملی آن‌ها باید در هزینه، انرژی، تأخیر یا نوع پردازش به‌روشنی نشان داده شود.

\begin{assignmentbox}{تمرین و ارزیابی مربوط به \eng{Predictive Coding}}
استاد اشاره کرد احتمالاً تمرینی دربارهٔ \eng{Predictive Coding} در سامانهٔ درس قرار می‌گیرد. همچنین با توجه به شرایط، ارزیابی ممکن است بیشتر به سمت امتحان کتبی مفهومی برود؛ یعنی توضیح مفاهیم، رسم معماری‌ها و تحلیل سناریوها.
\end{assignmentbox}

\chapter{توجه انتخابی}

\section{حافظهٔ حسی و گلوگاه توجه}
محیط حجم عظیمی از اطلاعات تولید می‌کند. حواس ما بخش زیادی از آن را در \term{حافظهٔ حسی}{Sensory Memory} ثبت می‌کنند، اما این اطلاعات فقط برای زمان کوتاه باقی می‌ماند. همهٔ آن‌ها نمی‌توانند وارد حافظهٔ کاری شوند. بنابراین باید انتخاب کنیم. این انتخاب همان \term{توجه انتخابی}{Selective Attention} است.

مسیر کلی:
\[
\text{محیط} \rightarrow \text{حافظهٔ حسی} \rightarrow \text{توجه انتخابی} \rightarrow \text{حافظهٔ کاری} \rightarrow \text{عمل}
\]

نکتهٔ اصلی این است که حافظهٔ حسی ظرفیت نسبتاً زیادی دارد، اما ماندگاری آن کوتاه است. وقتی وارد یک محیط می‌شویم، صداها، تصویرها، بوها و حس‌های بدنی زیادی هم‌زمان ثبت می‌شوند. با این حال، فقط بخشی از آن‌ها از گلوگاه توجه عبور می‌کند و وارد پردازش آگاهانه یا حافظهٔ کاری می‌شود. مسئلهٔ نظریه‌های توجه این است که این انتخاب در کجای مسیر و بر اساس چه چیزی انجام می‌شود.

این بحث با \eng{attention} در \eng{Transformer} یکی نیست. در مدل زبانی، معمولاً \eng{prompt} از قبل محیط مدل را تعیین کرده است. اما در سیستم‌هایی که واقعاً در محیط فیزیکی‌اند، مثل دستیار صوتی، دوربین یا ربات، سیستم باید از میان سیگنال‌های بیرونی تشخیص دهد کدام بخش اصلاً ارزش پردازش عمیق دارد.

\section{\eng{Broadbent's Filter Theory}: انتخاب زودهنگام}
در \term{انتخاب زودهنگام}{Early Selection}، فیلتر قبل از درک معنا عمل می‌کند. ورودی‌ها بر اساس ویژگی‌های فیزیکی یا آکوستیکی مثل شدت، جهت، آشنایی صدا یا نزدیکی انتخاب می‌شوند. سپس فقط ورودی انتخاب‌شده وارد پردازش معنایی می‌شود.

این نظریه توضیح می‌دهد چرا در یک مهمانی می‌توانیم روی صدای یک نفر تمرکز کنیم و بسیاری از صداهای دیگر را نفهمیم.

مسیر سادهٔ \eng{Early Selection} چنین است:
\[
\text{\lr{Multiple Inputs}} \rightarrow \text{\lr{Physical Filter}} \rightarrow \text{\lr{Semantic Processing}}
\]
برای مثال، در مهمانی ممکن است صدای دوست روبه‌روی خود را انتخاب کنیم چون نزدیک‌تر است، جهت صدا مشخص است یا صدایش برای ما آشناست. در این دیدگاه، ابتدا کانال صوتی انتخاب می‌شود و سپس معنای همان کانال پردازش می‌شود.

محدودیت نظریه این است که گاهی اطلاعات ظاهراً انتخاب‌نشده هم اثر می‌گذارند. اگر مشغول صحبت با کسی باشیم اما فرد دیگری در گوشهٔ اتاق اسم ما را صدا بزند، ممکن است ناگهان متوجه شویم. اگر پیام‌های دیگر کاملاً حذف شده باشند، توضیح این پدیده دشوار می‌شود.

\section{\eng{Late Selection}}
در \term{انتخاب دیرهنگام}{Late Selection}، ذهن ابتدا ورودی‌ها را تا حدی از نظر معنا پردازش می‌کند و سپس انتخاب می‌کند کدام ورودی مهم‌تر است. این نظریه پدیده‌ای مثل شنیدن اسم خود در یک مهمانی شلوغ را بهتر توضیح می‌دهد.

مشکل این دیدگاه آن است که پردازش معنایی همهٔ ورودی‌ها هزینهٔ بسیار زیادی دارد.

در \eng{Late Selection}، فیلتر پس از پردازش معنایی نسبی قرار دارد:
\[
\text{\lr{Multiple Inputs}} \rightarrow \text{\lr{Semantic Processing}} \rightarrow \text{\lr{Selection}}
\]
پس اگر اسم خودمان در کانالی غیر از مکالمهٔ اصلی شنیده شود، چون معنای آن برای ما مهم است، می‌تواند انتخاب شود. اما اگر این نظریه را خیلی قوی بگیریم، باید بپذیریم مغز تقریباً همهٔ ورودی‌ها را از نظر معنا پردازش می‌کند؛ چیزی که با محدودیت انرژی و تجربهٔ روزمره سازگار نیست.

\section{\eng{Perceptual Load Theory}}
\term{نظریهٔ بار ادراکی}{Perceptual Load Theory} می‌گوید هر دو حالت وجود دارند و بسته به بار ادراکی \eng{task} فعال می‌شوند.

\begin{itemize}
  \item بار ادراکی بالا: \eng{early selection} غالب است.
  \item بار ادراکی پایین: \eng{late selection} امکان‌پذیرتر است.
\end{itemize}

مثلاً رانندگی در جادهٔ ناآشنا و شب، بار ادراکی بالایی دارد و مکالمهٔ اطراف را کمتر پردازش می‌کنیم. اما در مسیر آشنا و ساده، ظرفیت بیشتری برای پردازش ورودی‌های دیگر داریم.

بنابراین اختلاف \eng{Early Selection} و \eng{Late Selection} را نباید فقط به شکل «کدام نظریه درست است؟» دید. پاسخ بهتر این است که سیستم توجه انسان انعطاف‌پذیر است. وقتی کار سخت، تازه، خطرناک یا پرجزئیات است، ذهن زودتر فیلتر می‌کند. وقتی کار ساده، تکراری یا آشناست، ظرفیت بیشتری برای پردازش معنایی ورودی‌های جانبی باقی می‌ماند.

\section{\eng{Treisman's Attenuation Theory}}
در نظریهٔ \term{تضعیف}{Attenuation}، فیلتر صفر و یکی نیست. اطلاعات حذف کامل نمی‌شوند؛ برخی تقویت و برخی تضعیف می‌شوند. محرک‌هایی با \term{آستانه}{Threshold} پایین، مثل اسم خود فرد، حتی اگر ضعیف باشند می‌توانند فعال شوند.

این ایده برای سیستم‌های صوتی نیز مهم است: \eng{wake word} مثل «\eng{Hey Siri}» یا «\eng{Alexa}» برای سیستم آستانهٔ پایینی دارد.

در این نظریه، فیلتر مثل دروازه‌ای نیست که بعضی پیام‌ها را کامل عبور دهد و بقیه را کامل حذف کند. بهتر است آن را مثل تنظیم‌کنندهٔ شدت ببینیم: پیام اصلی تقویت می‌شود، پیام‌های دیگر تضعیف می‌شوند، اما اگر محرکی اهمیت یا آشنایی زیادی داشته باشد، با وجود تضعیف می‌تواند از آستانه عبور کند. اسم خود فرد یا \eng{wake word} دقیقاً چنین جایگاهی دارد.

\section{پردازش سری و موازی}
بسیاری از کارهای ساده می‌توانند هم‌زمان انجام شوند، اما کارهای پیچیده‌ای مثل دو محاسبهٔ دشوار معمولاً واقعاً موازی انجام نمی‌شوند؛ ما بیشتر بین آن‌ها \eng{task switching} می‌کنیم. بنابراین توجه به یک گلوگاه نیاز دارد که تعیین کند کدام ورودی در هر لحظه پردازش عمیق‌تر شود.

مثال کلاس این بود که اگر از فرد بخواهیم هم‌زمان دو ضرب دشوار انجام دهد، معمولاً واقعاً هر دو را موازی حل نمی‌کند؛ بلکه سریع بین آن‌ها جابه‌جا می‌شود. کارهای ساده‌تر ممکن است هم‌زمان به نظر برسند، اما پردازش‌هایی که نیازمند معنا، تصمیم یا محاسبه‌اند معمولاً به انتخاب و ترتیب نیاز دارند. همین نیاز، اهمیت توجه انتخابی را در معماری شناختی نشان می‌دهد.

\chapter{پردازش گفتار و نگاشت توجه انتخابی به ماشین}

\section{دستیار صوتی و \eng{Wake Word}}
یک دستیار صوتی همیشه در محیط واقعی قرار دارد. اگر بخواهد همهٔ صداها را دائماً به متن تبدیل کند و معنای همه را بفهمد، هزینهٔ محاسباتی و انرژی بسیار زیاد می‌شود. بنابراین معمولاً دو مرحله دارد:

\begin{itemize}
  \item \textbf{مرحلهٔ \eng{Early Selection}}: تشخیص \eng{wake word} با پردازش سبک آکوستیکی.
  \item \textbf{مرحلهٔ \eng{Late Selection}}: پس از فعال‌شدن، تبدیل گفتار به متن، \eng{NLU}، تشخیص \eng{intent} و \eng{slot filling}.
\end{itemize}

مثال:
\[
\text{\lr{Speech}} \rightarrow \text{\lr{Wake Word Detection}} \rightarrow \text{\lr{ASR}} \rightarrow \text{\lr{NLU}} \rightarrow \text{\lr{Action}}
\]

\section{تولید گفتار انسان}
گفتار از تعامل دیافراگم، ریه، نای، حنجره، تارهای صوتی، زبان، دهان، بینی، فک و دندان‌ها تولید می‌شود. موج صوتی علاوه بر محتوای زبانی، اطلاعاتی دربارهٔ هویت گوینده، جنسیت، لهجه، احساسات و ویژگی‌های فیزیکی مجرای صوتی دارد.

در فارسی معمولاً شش واکهٔ اصلی داریم: سه واکهٔ کوتاه و سه واکهٔ بلند. تعداد واکه‌ها در زبان‌هایی مثل انگلیسی بیشتر و وابسته به لهجه است.

\section{از موج خام تا ویژگی}
\subsection{\eng{Raw Wave}}
موج صوتی خام را می‌توان روی دو محور دید: زمان و دامنه. دامنهٔ بزرگ‌تر معمولاً صدای بلندتر را نشان می‌دهد و سرعت نوسان موج با فرکانس مرتبط است. ماشین با موج آنالوگ کار نمی‌کند؛ بنابراین موج باید به دنباله‌ای از عددها تبدیل شود تا بتوان آن را به مدل داد.

\subsection{\eng{Sampling}}
ماشین با موج آنالوگ کار نمی‌کند؛ باید سیگنال نمونه‌برداری و دیجیتال شود. طبق قضیهٔ \eng{Nyquist}، نرخ نمونه‌برداری باید دست‌کم دو برابر بیشترین فرکانس موردنظر باشد.

\subsection{\eng{Framing} و \eng{Windowing}}
سیگنال صوتی به \eng{frame}های کوتاه تقسیم می‌شود. معمولاً \eng{frame}ها هم‌پوشانی دارند تا تغییرات مهم در مرز \eng{frame}ها از دست نرود. \eng{Hamming Window} به مرکز \eng{frame} وزن بیشتر و به لبه‌ها وزن کمتر می‌دهد.

\subsection{\eng{Fourier Transform} و \eng{Spectrogram}}
تبدیل فوریه نشان می‌دهد چه فرکانس‌هایی در سیگنال حضور دارند. با اعمال این تبدیل روی \eng{frame}های کوتاه، تصویری به نام \term{اسپکتروگرام}{Spectrogram} ساخته می‌شود. اسپکتروگرام تغییرات انرژی فرکانسی را در طول زمان نشان می‌دهد.

در اسپکتروگرام، یک محور زمان، محور دیگر فرکانس و رنگ یا شدت تصویر انرژی هر فرکانس را نشان می‌دهد. گفتار، تنفس و نویز معمولاً الگوهای متفاوتی در اسپکتروگرام دارند؛ به همین دلیل حتی پیش از تبدیل گفتار به متن می‌توان برخی طبقه‌بندی‌های آکوستیکی را انجام داد.

\subsection{\eng{Formants}}
\term{فورمنت}{Formant} ناحیه‌ای از انرژی فرکانسی قوی در گفتار است. فورمنت‌ها با محدودهٔ فرکانسی نام‌گذاری می‌شوند، نه صرفاً بر اساس اولین نوار قابل مشاهده. بنابراین اگر باند پایین دیده نشود، نوار بالاتر الزاماً \eng{F1} نیست.

فورمنت‌ها با شکل مجرای صوتی و کیفیت واکه‌ها مرتبط‌اند و از \eng{pitch} یا فرکانس پایه جدا هستند. به‌طور کلی صدای مردان معمولاً انرژی بیشتری در فرکانس‌های پایین‌تر دارد و صدای زنان معمولاً زیرتر است، اما تشخیص گوینده یا جنسیت را نباید فقط به یک باند ساده فروکاست؛ لهجه، سن، میکروفون و شرایط ضبط نیز اثر دارند.

\section{\eng{MFCC}}
\term{ضرایب کپسترال فرکانس مل}{Mel-Frequency Cepstral Coefficients / MFCC} نمایش عددی مفیدی از صوت است. مسیر کلی:
\[
\text{\lr{Waveform}} \rightarrow \text{\lr{Frames}} \rightarrow \text{\lr{Window}} \rightarrow \text{\lr{Fourier}} \rightarrow \text{\lr{Mel Filter Bank}} \rightarrow \text{\lr{MFCC}}
\]

\eng{Mel Filter Bank} با حساسیت شنوایی انسان سازگارتر است: در فرکانس‌های پایین تفکیک بیشتری نگه می‌دارد و در فرکانس‌های بالا رزولوشن کمتر می‌شود. فیلترهای مل معمولاً مثلثی و هم‌پوشان‌اند. در محاسبهٔ رایج \eng{MFCC}، پس از انرژی‌های \eng{mel}، لگاریتم و سپس \eng{DCT} اعمال می‌شود تا ضرایب فشرده‌تر و مناسب‌تری برای مدل‌های کلاسیک به دست آید.

\section{کاربرد در طبقه‌بندی}
برای تشخیص \eng{wake word} یا حتی طبقه‌بندی‌هایی مثل تشخیص الگوهای سرفه، می‌توان از \eng{MFCC} یا \eng{spectrogram} ویژگی استخراج کرد و سپس با معیارهایی مثل \eng{similarity}، \eng{correlation} یا یک \eng{classifier} بررسی کرد آیا دو گروه قابل تفکیک‌اند یا نه.

در مثال پزشکی کلاس، اگر بخواهیم از صدای سرفه یا تنفس افراد مبتلا و غیرمبتلا ویژگی بگیریم، ابتدا باید نشان دهیم \eng{feature}ها واقعاً تفاوت قابل اتکا دارند. اگر همبستگی یا شباهت دو گروه بسیار بالا باشد، \eng{classifier} احتمالاً اطلاعات کافی برای تفکیک ندارد. بنابراین استخراج ویژگی فقط گام اول است؛ اعتبار ویژگی باید با داده و ارزیابی روشن شود.

\begin{assignmentbox}{تمرین \eng{Early Selection}}
یکی از تمرین‌های اعلام‌شده: یک فایل صوتی کوتاه شامل اسم خود و یک \eng{wake-up phrase} ضبط شود، یک فایل نویز محیطی تعریف شود و ویژگی‌های \eng{MFCC} استخراج گردد. مهلت کلی تمرین‌های مطرح‌شده تا پایان اردیبهشت اعلام شد.
\end{assignmentbox}

\chapter{مدل‌های گفتاری: \eng{wav2vec}، \eng{HuBERT} و \eng{Whisper}}

\section{آیا همیشه باید گفتار به متن تبدیل شود؟}
معماری رایج:
\[
\text{\lr{Speech}} \rightarrow \text{\lr{Text}} \rightarrow \text{\lr{NLP Task}} \rightarrow \text{\lr{Action}}
\]

اما از نظر نظری همیشه لازم نیست متن میانی داشته باشیم. برای \eng{task}های محدود می‌توان مستقیماً از \eng{waveform}، \eng{spectrogram}، \eng{MFCC} یا \eng{embedding} صوتی به \eng{intent} یا \eng{class} رسید. با این حال، متن مزیت دارد: دادهٔ آموزشی متنی فراوان‌تر است، اتصال به \eng{API} و پایگاه داده ساده‌تر است و \eng{ambiguity} برخی فرمان‌ها کمتر می‌شود.

در کلاس این پرسش مطرح شد که اگر هدف فقط فهم فرمان است، چرا اصلاً گفتار را به متن تبدیل کنیم؟ پاسخ این است که حذف متن ممکن است در مسئله‌های محدود مفید باشد، اما در سیستم عمومی معمولاً متن یک لایهٔ میانی استاندارد و قابل اتصال می‌سازد. بسیاری از ابزارهای پایین‌دستی، پایگاه‌های داده، \eng{API}ها و مدل‌های زبانی با متن راحت‌تر کار می‌کنند. علاوه بر آن، دادهٔ متنیِ برچسب‌دار بسیار بیشتر از صوت برچسب‌دار دقیق است.

\section{سه جزء سیستم گفتاری}
\begin{itemize}
  \item \term{تشخیص خودکار گفتار}{ASR}: تبدیل گفتار به متن.
  \item \term{فهم زبان طبیعی}{NLU}: استخراج \eng{intent}، \eng{slot} و معنا از متن یا \eng{representation}.
  \item \term{تبدیل متن به گفتار}{TTS}: تولید پاسخ صوتی از متن.
\end{itemize}

\section{ویژگی‌های \eng{deterministic} و \eng{learnable}}
\eng{Spectrogram} و \eng{MFCC} معمولاً با فرمول‌های ریاضی \eng{deterministic} تولید می‌شوند و مرحلهٔ آموزشی ندارند. در مقابل، می‌توان \eng{waveform} خام را به \eng{CNN} داد تا \eng{feature}ها را یاد بگیرد:
\[
\text{\lr{Raw Waveform}} \rightarrow \text{\lr{CNN}} \rightarrow \text{\lr{Learned Features}}
\]

در \eng{CNN}، فیلترها یا \eng{kernel}ها پارامترهای قابل یادگیری‌اند. در لایه‌های اولیه، مدل معمولاً الگوهای محلی و ساده‌تر را می‌گیرد و در لایه‌های عمیق‌تر بازنمایی‌ها انتزاعی‌تر می‌شوند. \eng{Pooling} می‌تواند بعد را کاهش دهد، اطلاعات برجسته را فشرده کند و هزینهٔ محاسباتی را پایین بیاورد. در \eng{wav2vec}، ورودی معمولاً \eng{waveform} خام است؛ اشاره به \eng{spectrogram} در کلاس بیشتر برای توضیح شهودی \eng{CNN} بود.

\section{\eng{Self-Supervised Learning} در گفتار}
دادهٔ صوتی خام زیاد است، اما صوت همراه با \eng{transcript} دقیق کمتر است. مدل‌های \eng{self-supervised} از خود داده \eng{task} آموزشی می‌سازند؛ مثلاً بخشی از فریم‌ها را \eng{mask} می‌کنند و مدل باید از \eng{context} آن‌ها را پیش‌بینی کند یا \eng{representation} مناسب بیاموزد.

\section{\eng{wav2vec 2.0}}
مسیر کلی \eng{wav2vec 2.0}:
\[
\begin{aligned}
\text{\lr{Raw Waveform}} &\rightarrow \text{\lr{CNN Feature Encoder}} \rightarrow \text{\lr{Transformer Encoder}}\\
&\rightarrow \text{\lr{Contextual Representations}}
\end{aligned}
\]

بخشی از \eng{representation}های نهفته \eng{mask} می‌شود. سپس مدل با هدف \eng{contrastive} یاد می‌گیرد \eng{representation} درست را از نمونه‌های منفی جدا کند. \eng{Quantization} با \eng{codebook} کمک می‌کند هدف آموزشی گسسته‌تر و پایدارتر باشد.

\subsection{\eng{Codebook} و \eng{Quantization}}
در صوت، برخلاف متن، \eng{vocabulary} طبیعی و گسسته نداریم. \eng{Quantization} یک بردار پیوسته را به نزدیک‌ترین بردار نماینده در \eng{codebook} نگاشت می‌کند. این کار شبیه ساختن \eng{vocabulary} صوتی است، اما نباید \eng{representation} غنی مسیر اصلی را خیلی زود نابود کند. در \eng{wav2vec 2.0}، \eng{quantized targets} عمدتاً برای \eng{loss} استفاده می‌شوند.

در هدف \eng{contrastive}، مدل باید نمایش درست مربوط به بخش \eng{masked} را به نمونهٔ مثبت نزدیک و از نمونه‌های منفی دور کند. بنابراین \eng{quantization} برای ساختن هدف آموزشی روشن‌تر به کار می‌رود، در حالی که \eng{Transformer Encoder} همچنان روی بازنمایی‌های پیوسته و غنی کار می‌کند. بعد از \eng{pretraining}، مدل می‌تواند با دادهٔ برچسب‌دار کمتر برای \eng{ASR}، طبقه‌بندی یا کارهای گفتاری دیگر \eng{fine-tune} شود.

مثال کلاس برای \eng{codebook}، مکان‌یابی سلسله‌مراتبی بود: نخست کشور، بعد شهر، بعد منطقه. این مثال نشان می‌دهد یک فضای پیوسته را می‌توان در چند سطح به واحدهای گسسته‌تر و قابل مقایسه‌تر تبدیل کرد.

\section{\eng{HuBERT}}
\eng{HuBERT} نیز از \eng{masking} استفاده می‌کند، اما معمولاً هدف آن پیش‌بینی برچسب‌های خوشه‌ای پنهان‌شده است. این تفاوت نشان می‌دهد همهٔ مدل‌های \eng{self-supervised} گفتاری یک \eng{objective} واحد ندارند.

\section{\eng{Whisper}}
\eng{Whisper} برخلاف \eng{wav2vec} که بیشتر برای \eng{pretraining representation} مطرح شد، مستقیماً صوت را به متن تولید می‌کند. دادهٔ آموزشی آن حدود ۶۸۰ هزار ساعت صوت-متن چندزبانه و پرتنوع است و به همین دلیل نسبت به لهجه، نویز و شرایط واقعی \eng{robust}تر است.

\subsection{چرا \eng{Encoder-Decoder}؟}
مدل باید هم صوت را بفهمد و هم متن تولید کند. بنابراین معماری مناسب:
\[
\text{\lr{Audio Encoder}} \rightarrow \text{\lr{Text Decoder}}
\]

در \eng{Whisper} ورودی معمولاً \eng{log-mel spectrogram} است. \eng{Encoder} با \eng{self-attention} کامل، \eng{embedding} صوتی \eng{contextualized} می‌سازد. \eng{Decoder} به‌صورت \eng{autoregressive} متن را تولید می‌کند.

دلیل نیاز به دو بخش این است که وظیفه فقط استخراج بازنمایی صوتی نیست. \eng{Encoder} باید از ورودی صوتی یک بازنمایی قابل استفاده بسازد؛ اما خروجی نهایی دنباله‌ای از \eng{token}های متنی است و باید یکی‌یکی تولید شود. بنابراین بخش \eng{Decoder} لازم است تا با توجه به متن تولیدشده تا آن لحظه و بازنمایی صوتی، \eng{token} بعدی را پیش‌بینی کند.

\subsection{\eng{Masked Self-Attention} و \eng{Cross-Attention}}
در \eng{Encoder}، کل ورودی صوتی موجود است، پس \eng{self-attention} ماسک ندارد. در \eng{Decoder}، توکن‌های آینده نباید دیده شوند، پس \eng{masked self-attention} لازم است. همچنین \eng{Decoder} با \eng{cross-attention} به خروجی \eng{Encoder} توجه می‌کند تا متن تولیدشده با صوت اصلی هم‌راستا بماند.

\eng{Cross-Attention} پل بین دو جهان است: \eng{Decoder} در فضای متن کار می‌کند، اما برای اینکه متن تولیدی از صوت جدا نشود باید به خروجی \eng{Encoder} نگاه کند. اگر این اتصال نباشد، \eng{Decoder} فقط شبیه یک مدل زبانی عمل می‌کند و ممکن است متن روان ولی نامرتبط با صوت تولید کند.

\section{ایدهٔ پژوهشی از توجه انتخابی}
اگر بخواهیم دستیار صوتی شناختی‌تر بسازیم، می‌توانیم بار ادراکی محیط را تخمین بزنیم و بر اساس آن بین پردازش سبک و سنگین سوئیچ کنیم:
\[
\text{\lr{High Perceptual Load}} \rightarrow \text{\lr{Early Selection}}
\]
\[
\text{\lr{Low Perceptual Load}} \rightarrow \text{\lr{Late Selection}}
\]
این ایده \eng{trade-off} مهندسی میان هزینهٔ محاسباتی و دقت معنایی را به زبان شناختی توضیح می‌دهد.

\chapter{سیستم پاداش و یادگیری تقویتی}

\section{پاداش در مغز}
پس از بحث مدل‌های شنیداری، کلاس وارد سیستم پاداش شد. در انسان، پاداش به‌صورت حس رضایت، لذت یا موفقیت تجربه می‌شود و با آزادسازی \eng{dopamine} ارتباط دارد. دوپامین در ناحیهٔ \term{VTA}{Ventral Tegmental Area} تولید می‌شود و در مسیرهای دوپامینرژیک اثر می‌گذارد.

یادگیری مبتنی بر پاداش را می‌توان با زنجیرهٔ سادهٔ زیر فهمید: سیستم در یک وضعیت قرار دارد، عملی انجام می‌دهد، وضعیت تغییر می‌کند و این تغییر با پاداش، تنبیه یا بازخورد خنثی همراه می‌شود. پاداش نقش \eng{reinforcer} دارد؛ یعنی احتمال تکرار رفتارهای مفید یا خوشایند را در آینده تغییر می‌دهد.

پاداش فقط به رخدادهای زیستی مثل غذا محدود نیست. موفقیت در یک کار، دریافت پیام مهم، یا دیدن نشانه‌ای که قبلاً با تجربهٔ خوب همراه بوده نیز می‌تواند سیستم پاداش را فعال کند. نکتهٔ مهم کلاس این بود که دوپامین معمولاً با یک \eng{transition} معنادار مرتبط است: تغییری در وضعیت که برای سیستم ارزش یا انتظار ایجاد می‌کند.

\section{دو مسیر مهم}
\subsection{\eng{Mesolimbic Pathway}}
از \eng{VTA} به سمت سیستم \eng{limbic} می‌رود و با رضایت، شرطی‌شدن، تقویت رفتار، میل به تکرار و اعتیاد مرتبط است. این مسیر به‌طور ساده می‌گوید: «این رفتار قبلاً خوب بوده؛ دوباره انجامش بده.»

برای مثال، اگر نوشیدن چای بعد از کار بارها تجربهٔ خوشایندی ایجاد کرده باشد، نشانه‌های پایان کار می‌توانند میل به چای را فعال کنند. این مسیر کمک می‌کند رفتارهای پاداش‌دهنده تکرار شوند، اما همین سازوکار در عادت‌های ناسالم و اعتیاد نیز نقش دارد.

\subsection{\eng{Mesocortical Pathway}}
از \eng{VTA} به \eng{Prefrontal Cortex} می‌رود و با تصمیم‌گیری، تحلیل پیامدها، کنترل رفتار و ارزیابی بلندمدت مرتبط است. اگر \eng{mesolimbic} نقش گاز داشته باشد، \eng{mesocortical} تا حدی نقش تحلیل‌گر و ترمز را دارد.

مثال کلاس کیک بود: مسیر پاداش ممکن است خوردن کیک را جذاب کند، اما بخش پیش‌پیشانی می‌تواند پیامدهای بلندمدت، رژیم، گرسنگی واقعی و ارزش انجام این عمل را بررسی کند. بنابراین سیستم پاداش فقط «لذت» نیست؛ با کنترل، ارزیابی و تصمیم‌گیری نیز تعامل دارد.

\section{ارتباط با \eng{Reinforcement Learning}}
در \term{یادگیری تقویتی}{Reinforcement Learning} عامل در یک \eng{state} قرار دارد، \eng{action} انجام می‌دهد، به \eng{state} جدید منتقل می‌شود و \eng{reward} می‌گیرد:
\[
s_t \xrightarrow{a_t} s_{t+1}, \qquad r_t
\]

یادگیری از طریق تعامل با محیط و \eng{feedback} انجام می‌شود. این تفاوت اصلی آن با \eng{supervised learning} است که در آن \eng{label} مستقیم داریم، و با \eng{unsupervised learning} که در آن مدل معمولاً ساختار داده را بدون \eng{action} و \eng{reward} یاد می‌گیرد.

\begin{definitionbox}{تفاوت سه نوع یادگیری}
\begin{itemize}
  \item \eng{Supervised Learning}: ورودی و \eng{label} داریم.
  \item \eng{Unsupervised Learning}: \eng{label} مستقیم نداریم و ساختار داده را پیدا می‌کنیم.
  \item \eng{Reinforcement Learning}: عامل با محیط تعامل می‌کند و از پاداش یا تنبیه یاد می‌گیرد.
\end{itemize}
\end{definitionbox}

در \eng{Supervised Learning}، مدل برای هر ورودی معمولاً پاسخ درست یا \eng{label} دارد. در \eng{Unsupervised Learning}، مدل بدون برچسب مستقیم، ساختار داده را پیدا می‌کند؛ مثلاً خوشه‌ها یا بازنمایی‌ها را یاد می‌گیرد. اما در \eng{Reinforcement Learning} مسئله فقط نبودن برچسب نیست. عامل باید عمل کند، پیامد عمل خود را ببیند و از \eng{reward} یاد بگیرد. بنابراین تجربه، ترتیب زمانی و تعامل با محیط بخش اصلی مسئله‌اند.

\chapter{پروژه، تمرین‌ها و جمع‌بندی مسیر پژوهشی}

\section{مسیر پژوهشی پیشنهادی درس}
یکی از مسیرهای پژوهشی مهم این است:
\begin{enumerate}
  \item یک ضعف شناختی در مدل نشان دهیم.
  \item بررسی کنیم آیا خطای مدل با خطای انسان \eng{correlation} دارد یا نه.
  \item آن ضعف را با یک نظریهٔ شناختی توضیح دهیم.
  \item \eng{dataset} یا آزمایش مناسب طراحی کنیم.
  \item مدل را با روش‌هایی مثل \eng{fine-tuning}، \eng{prompting}، \eng{RAG}، تغییر معماری یا اضافه کردن ماژول شناختی بهبود دهیم.
  \item نشان دهیم بهبود فقط افزایش سطحی عملکرد نیست، بلکه از نظر شناختی معنا دارد.
\end{enumerate}

سطح ورود به معماری‌ها در کلاس عمداً بین دو حد نگه داشته شد: نه آن‌قدر شناختی که از هوش مصنوعی دور شود، و نه آن‌قدر فنی که درس به تکرار جزئیات \eng{CNN}، \eng{Transformer} یا \eng{contrastive learning} تبدیل شود. هدف این بود که دانشجو بتواند نظریهٔ شناختی را بشناسد، مدل محاسباتی متناظر را در سطح لازم بفهمد و از آن برای طراحی ایدهٔ پژوهشی استفاده کند.

\section{نمونه‌های موضوع پروژه}
\begin{itemize}
  \item \eng{dataset} شناختی برای مقایسهٔ انسان و مدل در \eng{reasoning}، حافظه، توجه یا زبان.
  \item مدل‌سازی \eng{cognitive load} در \eng{LLM} و طراحی \eng{prompt}های پرنویز.
  \item بررسی \eng{chunking} در \eng{task}های \eng{long-horizon agent}.
  \item کاهش \eng{bias} در \eng{LLM} با چارچوب \eng{predictive coding}.
  \item طراحی سیستم \eng{wake-word detection} با \eng{MFCC} و نویز محیطی.
  \item مقایسهٔ \eng{early/late selection} در معماری دستیار صوتی.
  \item بررسی یا شبیه‌سازی مفهومی پردازنده‌های \term{نورومورفیک}{Neuromorphic}.
\end{itemize}

\begin{assignmentbox}{پروژهٔ پایانی}
\begin{itemize}
  \item پروژه باید مرتبط با مباحث کلاس باشد.
  \item پروژه حداکثر دو نفره است.
  \item هر گروه باید \eng{proposal} حدوداً یک صفحه‌ای ارائه دهد.
  \item استاد پس از بررسی ممکن است \eng{scope} پروژه را اصلاح کند.
  \item زمان تحویل پروژه حدوداً پیش از امتحان و نزدیک به پیش از ۲۰ خرداد مطرح شد.
  \item اگر دانشجویی ایده نداشته باشد، استاد می‌تواند موضوع پیشنهاد دهد.
  \item یکی از موضوع‌های پیشنهادی، بررسی یا پیاده‌سازی مجازی یک پردازندهٔ \eng{Neuromorphic} بود.
\end{itemize}
\end{assignmentbox}

\begin{assignmentbox}{میان‌ترم و تمرین‌ها}
\begin{itemize}
  \item تا جلسهٔ دهم، سه تمرین مطرح شده بود و مهلت کلی آن‌ها پایان اردیبهشت اعلام شد.
  \item تمرین چهارم قرار بود از بحث \eng{Reinforcement Learning} اعلام شود.
  \item میان‌ترم از مباحث کلاس است و می‌تواند شامل توضیح مفاهیم، رسم معماری‌ها، حل مسئله یا تحلیل سناریو باشد.
  \item زمان پیشنهادی میان‌ترم حوالی هفتهٔ سوم اردیبهشت بود و قرار شد در گروه کلاس دربارهٔ تاریخ آن نظرسنجی شود.
  \item استاد گفت وضعیت ضبط آفلاین، دسترسی به \eng{VPN}/\eng{proxy} دانشگاه و گزینه‌های تاریخ میان‌ترم را در گروه پیگیری می‌کند.
\end{itemize}
\end{assignmentbox}

\begin{assignmentbox}{نکتهٔ پایانی دربارهٔ ساختار درس}
استاد اشاره کرد این درس برای نخستین بار در دانشکده با این سیلابس ارائه می‌شود و هنوز ساختار دستیار آموزشی کاملاً تثبیت نشده است. از دانشجویان علاقه‌مند و فعال خواسته شد برای همکاری احتمالی به‌عنوان \eng{TA} در ترم‌های آینده اعلام آمادگی کنند.
\end{assignmentbox}

\section{جمع‌بندی کل درس تا اینجا}
درس از این ایده شروع شد که مدل‌های هوش مصنوعی، حتی مدل‌های بزرگ، هنوز در برخی مهارت‌های شناختی مانند استدلال عمیق، قیاس، توجه، حافظهٔ کاری، سوگیری و تعامل با محیط ضعف دارند. راه‌حل صرفاً بزرگ‌تر کردن مدل یا اضافه کردن داده نیست. باید بفهمیم انسان این مهارت‌ها را چگونه انجام می‌دهد و کدام نظریهٔ شناختی می‌تواند الهام‌بخش یک مدل محاسباتی بهتر باشد.

مسیر طی‌شده در جلسات موجود چنین بود:
\begin{itemize}
  \item تعریف علوم شناختی محاسباتی و نسبت آن با AI.
  \item نظریه‌های پردازش گفتار مانند \eng{Cohort} و \eng{TRACE}.
  \item معماری شناختی و \eng{ACT-R}.
  \item حافظهٔ کاری، حافظهٔ بلندمدت، \eng{chunking} و \eng{cognitive load}.
  \item \eng{attention} در \eng{Transformer} و محدودیت‌های \eng{context}.
  \item \eng{cognitive bias} و نظریه‌های \eng{Dual Process}، \eng{Cognitive Load} و \eng{Predictive Coding}.
  \item \eng{attention} انتخابی و نظریه‌های \eng{Early Selection}، \eng{Late Selection}، \eng{Perceptual Load} و \eng{Attenuation}.
  \item پردازش گفتار، \eng{MFCC}، مدل‌های \eng{self-supervised}، \eng{wav2vec}، \eng{HuBERT} و \eng{Whisper}.
  \item سیستم پاداش و ارتباط آن با \eng{Reinforcement Learning}.
\end{itemize}
